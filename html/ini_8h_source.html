67 /* Parse given INI-style file. May have [section]s, name=value pairs
68 (whitespace stripped), and comments starting with ';' (semicolon). Section
69 is "" if name=value pair parsed before any section heading. name:value
70 pairs are also supported as a concession to Python's configparser.
71
72 For each name=value pair parsed, call handler function with given user
73 pointer as well as section, name, and value (data only valid for duration
74 of handler call). Handler should return nonzero on success, zero on error.
75
76 Returns 0 on success, line number of first error on parse error (doesn't
77 stop on first error), -1 on file open error, or -2 on memory allocation
78 error (only when INI_USE_STACK is zero).
79 */
80 INI_API int ini_parse(const char *filename, ini_handler handler, void *user);
81
82 /* Same as ini_parse(), but takes a FILE* instead of filename. This doesn't
83 close the file when it's finished -- the caller must do that. */
84 INI_API int ini_parse_file(FILE *file, ini_handler handler, void *user);
85
86 /* Same as ini_parse(), but takes an ini_reader function pointer instead of
87 filename. Used for implementing custom or string-based I/O (see also
88 ini_parse_string). */
89 INI_API int ini_parse_stream(ini_reader reader, void *stream, ini_handler handler, void *user);
90
91 /* Same as ini_parse(), but takes a zero-terminated string with the INI data
92 instead of a file. Useful for parsing INI data from a network socket or
93 already in memory. */
